c = int(a - 100)

# Если вес пользователя больше оптимального, выводится сообщение о необходимости похудения на определенное количество килограммов
if b > c:
  print("You need to lose weight by", b - a + 100, "kg.")

# Если вес пользователя меньше оптимального, выводится сообщение о необходимости набрать определенное количество килограммов
elif b < c:
  print("You need to get better at", a - 100 - b, "kg.")

# Если вес пользователя соответствует оптимальному, выводится сообщение об этом
else:
   print("You have the optimal weight! ")

if a + b <= c or a + c <= b or b + c <= a:
  print("The triangle does not exist")

elif a == b and a == c:
  print("The triangle is equilateral")

# Проверка условия на равнобедренность треугольника и вывод сообщения, если треугольник равнобедренный
elif a == b or a == c or b == c:
  print("The triangle is isosceles")

# Вывод сообщения о том, что треугольник разносторонний (если не выполняются предыдущие условия)
else:
  print("The triangle is versatile")

# Считываем стоимость покупки с клавиатуры
==================================================================================================================================================================================================================

purchase_amount = float(input("Введите стоимость покупки (в рублях): "))

# Инициализируем переменную для хранения суммы скидки
discount = 0

# Проверяем условия для предоставления скидки и вычисляем её
if purchase_amount > 1000:
    discount = 0.05  # 5% скидка
elif purchase_amount > 500:
    discount = 0.03  # 3% скидка

# Вычисляем стоимость покупки с учетом скидки
discounted_amount = purchase_amount - (purchase_amount * discount)

# Выводим результат
print(f"Сумма покупки: {purchase_amount} руб.")

print(f"Итоговая стоимость с учетом скидки: {discounted_amount} руб.")

(ладья бьет по горизонтали и вертикали, слон – по диагоналям). 
Проверить, есть ли угроза королю и если есть, то от кого именно. 
Учесть возможность защиты (например, ладья не бьет через слона).
===============================================================================================================================================================================================

# Координаты короля, ладьи и слона на шахматной доске
king_x = int(input("Введите координату X короля: "))
king_y = int(input("Введите координату Y короля: "))

rook_x = int(input("Введите координату X ладьи: "))
rook_y = int(input("Введите координату Y ладьи: "))

bishop_x = int(input("Введите координату X слона: "))
bishop_y = int(input("Введите координату Y слона: "))

# Функция для проверки, может ли ладья бить короля
def is_rook_threatening(king_x, king_y, rook_x, rook_y):
    return king_x == rook_x or king_y == rook_y

# Функция для проверки, может ли слон бить короля
def is_bishop_threatening(king_x, king_y, bishop_x, bishop_y):
    return abs(king_x - bishop_x) == abs(king_y - bishop_y)

# Проверяем, есть ли угроза королю
rook_threatens_king = is_rook_threatening(king_x, king_y, rook_x, rook_y)
bishop_threatens_king = is_bishop_threatening(king_x, king_y, bishop_x, bishop_y)

if rook_threatens_king and not (rook_x == bishop_x or rook_y == bishop_y):
    print("Ладья угрожает королю.")
elif bishop_threatens_king:
    print("Слон угрожает королю.")
else:
    print("Нет угрозы королю.")
